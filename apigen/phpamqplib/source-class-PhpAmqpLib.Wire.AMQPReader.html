134:      * Wait until some data is retrieved from the socket.
135:      *
136:      * AMQPTimeoutException can be raised if the timeout is set
137:      *
138:      * @throws \PhpAmqpLib\Exception\AMQPTimeoutException
139:      */
140:     protected function wait()
141:     {
142:         if ($this->timeout == 0) {
143:             return;
144:         }
145: 
146:         // wait ..
147:         list($sec, $usec) = MiscHelper::splitSecondsMicroseconds($this->timeout);
148:         $result = $this->io->select($sec, $usec);
149: 
150:         if ($result === false) {
151:             throw new AMQPRuntimeException(sprintf("An error occurs", $this->timeout));
152:         }
153: 
154:         if ($result === 0) {
155:             throw new AMQPTimeoutException(sprintf("A timeout of %ds occurs while waiting for incoming data", $this->timeout));
156:         }
157:     }

411:     /**
412:      * Reads the next value as the provided field type.
413:      *
414:      * @param string $fieldType the char field type
415:      *
416:      * @return mixed
417:      */
418:     public function read_value($fieldType)
419:     {
420:         $this->bitcount = $this->bits = 0;
421: 
422:         $val = NULL;
423:         switch ($fieldType) {
424:             case 'S': // Long string
425:                 $val = $this->read_longstr();
426:                 break;
427:             case 'I': // Signed 32-bit
428:                 $val = $this->read_signed_long();
429:                 break;
430:             case 'D': // Decimal
431:                 $e = $this->read_octet();
432:                 $n = $this->read_signed_long();
433:                 $val = new AMQPDecimal($n, $e);
434:                 break;
435:             case 't':
436:                 $val = $this->read_octet();
437:                 break;
438:             case 'l':
439:                 $val = $this->read_longlong();
440:                 break;
441:             case 'T': // Timestamp
442:                 $val = $this->read_timestamp();
443:                 break;
444:             case 'F': // Table
445:                 $val = $this->read_table();
446:                 break;
447:             case 'A': // Array
448:                 $val = $this->read_array();
449:                 break;
450:             case 'b': // bit
451:                 $val = $this->read_bit();
452:                 break;
453:             case 'V': // void
454:                 $val = null;
455:                 break;                
456:             default:
457:                 // UNKNOWN TYPE
458:                 throw new AMQPRuntimeException("Usupported table field type {$fieldType}");
459:                 break;
460:         }

462:         return $val;
463:     }
